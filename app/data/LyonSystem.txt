-- LyonSystem.db

CREATE TABLE "NuevoIngreso" (
	"id_NuevoIngreso"	INTEGER NOT NULL UNIQUE,
	"nombre_guia"	TEXT NOT NULL,
	"nombre_empresa"	TEXT NOT NULL,
	"cantidad_productos"	INTEGER NOT NULL,
	"fecha_guia"	TEXT NOT NULL,
	"save_data"	TEXT NOT NULL,
	"path_guia"	TEXT NOT NULL,
	"path_factura"	TEXT NOT NULL,
	"nombre_producto"	TEXT NOT NULL,
	"descripcion_producto"	TEXT NOT NULL,
	"series_producto"	TEXT NOT NULL,
	PRIMARY KEY("id_NuevoIngreso" AUTOINCREMENT)
)

-- Tabla para el nuevo registro
CREATE TABLE "nuevo_registro" (
	"id_nuevo_registro"	INTEGER NOT NULL UNIQUE,
	"nombre_guia"	TEXT NOT NULL,
	"nombre_empresa"	TEXT NOT NULL,
	"cantidad_productos"	INTEGER NOT NULL,
	"fecha_guia"	TEXT NOT NULL,
	"save_data"	TEXT NOT NULL,
	"path_guia"	TEXT NOT NULL,
	"path_factura"	TEXT NOT NULL,
	PRIMARY KEY("id_nuevo_registro" AUTOINCREMENT)
)

-- Tabla para los productos
CREATE TABLE "productos" (
	"id_productos"	INTEGER NOT NULL UNIQUE,
	"nombre"	TEXT NOT NULL,
	"descripcion"	TEXT NOT NULL,
	"id_nuevo_registro"	INTEGER NOT NULL,
	PRIMARY KEY("id_productos" AUTOINCREMENT),
	FOREIGN KEY("id_nuevo_registro") REFERENCES "nuevo_registro"("id_nuevo_registro")
)

-- Tabla para las series
CREATE TABLE "series" (
	"id_series"	INTEGER NOT NULL UNIQUE,
	"serie"	TEXT NOT NULL,
	"id_productos"	INTEGER NOT NULL,
	PRIMARY KEY("id_series" AUTOINCREMENT),
	FOREIGN KEY("id_productos") REFERENCES "productos"("id_productos")
)
